guards.js

A jQuery plugin to
guard your forms with class.
(or any other selector) 

$.guards.style();
$.liveGuard("form");

$.guard(".required").using("required");

<input type="text" class="required" />

$.guard(".email").using("email");

<input type="text" class="required email" />

$.guard("#true").using(function(value, element) {
  return value === "true";
}).message("Please enter true!");

<input type="text" id="true" />
